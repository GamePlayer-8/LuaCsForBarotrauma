93  public static List<PriceInfo> CreatePriceInfos(XElement element, out PriceInfo defaultPrice)
94  {
95  var priceInfos = new List<PriceInfo>();
96  defaultPrice = null;
97  int basePrice = element.GetAttributeInt("baseprice", 0);
98  int minAmount = GetMinAmount(element);
99  int maxAmount = GetMaxAmount(element);
100  int minLevelDifficulty = element.GetAttributeInt("minleveldifficulty", 0);
101  bool canBeSpecial = element.GetAttributeBool("canbespecial", true);
102  float buyingPriceMultiplier = element.GetAttributeFloat("buyingpricemultiplier", 1f);
103  bool displayNonEmpty = element.GetAttributeBool("displaynonempty", false);
104  bool soldByDefault = element.GetAttributeBool("sold", element.GetAttributeBool("soldbydefault", true));
105  bool requiresUnlock = element.GetAttributeBool("requiresunlock", false);
106  foreach (XElement childElement in element.GetChildElements("price"))
107  {
108  float priceMultiplier = childElement.GetAttributeFloat("multiplier", 1.0f);
109  bool sold = childElement.GetAttributeBool("sold", soldByDefault);
110  int storeMinLevelDifficulty = childElement.GetAttributeInt("minleveldifficulty", minLevelDifficulty);
111  float storeBuyingMultiplier = childElement.GetAttributeFloat("buyingpricemultiplier", buyingPriceMultiplier);
112  string backwardsCompatibleIdentifier = childElement.GetAttributeString("locationtype", "");
113  if (!string.IsNullOrEmpty(backwardsCompatibleIdentifier))
114  {
115  backwardsCompatibleIdentifier = $"merchant{backwardsCompatibleIdentifier}";
116  }
117  string storeIdentifier = childElement.GetAttributeString("storeidentifier", backwardsCompatibleIdentifier);
118  // TODO: Add some error messages if we have defined the min or max amount while the item is not sold
119  var priceInfo = new PriceInfo(price: (int)(priceMultiplier * basePrice),
120  canBeBought: sold,
121  minAmount: sold ? GetMinAmount(childElement, minAmount) : 0,
122  maxAmount: sold ? GetMaxAmount(childElement, maxAmount) : 0,
123  canBeSpecial: canBeSpecial,
124  minLevelDifficulty: storeMinLevelDifficulty,
125  buyingPriceMultiplier: storeBuyingMultiplier,
126  displayNonEmpty: displayNonEmpty,
127  requiresUnlock: requiresUnlock,
128  storeIdentifier: storeIdentifier);
129  priceInfo.LoadReputationRestrictions(childElement);
130  priceInfos.Add(priceInfo);
131  }
132  bool soldElsewhere = soldByDefault && element.GetAttributeBool("soldelsewhere", element.GetAttributeBool("soldeverywhere", false));
133  defaultPrice = new PriceInfo(price: basePrice,
134  canBeBought: soldElsewhere,
135  minAmount: soldElsewhere ? minAmount : 0,
136  maxAmount: soldElsewhere ? maxAmount : 0,
137  canBeSpecial: canBeSpecial,
138  minLevelDifficulty: minLevelDifficulty,
139  buyingPriceMultiplier: buyingPriceMultiplier,
140  displayNonEmpty: displayNonEmpty,
141  requiresUnlock: requiresUnlock);
142  defaultPrice.LoadReputationRestrictions(element);
143  return priceInfos;
144  }
