77  protected void ReadConnectionInitializationStep(PendingClient pendingClient, IReadMessage inc, ConnectionInitialization initializationStep)
78  {
79  pendingClient.TimeOut = NetworkConnection.TimeoutThreshold;
80 
81  if (pendingClient.InitializationStep != initializationStep) { return; }
82 
83  pendingClient.UpdateTime = Timing.TotalTime + Timing.Step;
84 
85  switch (initializationStep)
86  {
87  case ConnectionInitialization.SteamTicketAndVersion:
88  var authPacket = INetSerializableStruct.Read<ClientSteamTicketAndVersionPacket>(inc);
89 
90  if (!Client.IsValidName(authPacket.Name, serverSettings))
91  {
92  RemovePendingClient(pendingClient, PeerDisconnectPacket.WithReason(DisconnectReason.InvalidName));
93  return;
94  }
95 
96  bool isCompatibleVersion =
97  Version.TryParse(authPacket.GameVersion, out var remoteVersion)
98  && NetworkMember.IsCompatible(remoteVersion, GameMain.Version);
99  if (!isCompatibleVersion)
100  {
101  RemovePendingClient(pendingClient, PeerDisconnectPacket.InvalidVersion());
102 
103  GameServer.Log($"{authPacket.Name} ({authPacket.SteamId}) couldn't join the server (incompatible game version)", ServerLog.MessageType.Error);
104  DebugConsole.NewMessage($"{authPacket.Name} ({authPacket.SteamId}) couldn't join the server (incompatible game version)", Microsoft.Xna.Framework.Color.Red);
105  return;
106  }
107 
108  pendingClient.Connection.Language = authPacket.Language.ToLanguageIdentifier();
109 
110  Client nameTaken = GameMain.Server.ConnectedClients.Find(c => Homoglyphs.Compare(c.Name.ToLower(), authPacket.Name.ToLower()));
111  if (nameTaken != null)
112  {
113  RemovePendingClient(pendingClient, PeerDisconnectPacket.WithReason(DisconnectReason.NameTaken));
114  GameServer.Log($"{authPacket.Name} ({authPacket.SteamId}) couldn't join the server (name too similar to the name of the client \"" + nameTaken.Name + "\").", ServerLog.MessageType.Error);
115  return;
116  }
117 
118  if (!pendingClient.AuthSessionStarted)
119  {
120  ProcessAuthTicket(authPacket, pendingClient);
121  }
122 
123  break;
124  case ConnectionInitialization.Password:
125  var passwordPacket = INetSerializableStruct.Read<ClientPeerPasswordPacket>(inc);
126 
127  if (pendingClient.PasswordSalt is null)
128  {
129  DebugConsole.ThrowError("Received password message from client without salt");
130  return;
131  }
132 
133  if (serverSettings.IsPasswordCorrect(passwordPacket.Password, pendingClient.PasswordSalt.Value))
134  {
135  pendingClient.InitializationStep = ConnectionInitialization.ContentPackageOrder;
136  }
137  else
138  {
139  pendingClient.Retries++;
140  if (serverSettings.BanAfterWrongPassword && pendingClient.Retries > serverSettings.MaxPasswordRetriesBeforeBan)
141  {
142  const string banMsg = "Failed to enter correct password too many times";
143  BanPendingClient(pendingClient, banMsg, null);
144  RemovePendingClient(pendingClient, PeerDisconnectPacket.Banned(banMsg));
145  return;
146  }
147  }
148 
149  pendingClient.UpdateTime = Timing.TotalTime;
150  break;
151  case ConnectionInitialization.ContentPackageOrder:
152  pendingClient.InitializationStep = ConnectionInitialization.Success;
153  pendingClient.UpdateTime = Timing.TotalTime;
154  break;
155  }
156  }

202  protected void UpdatePendingClient(PendingClient pendingClient)
203  {
204  var skipRemove = false;
205  var result = GameMain.LuaCs.Hook.Call<bool?>("handlePendingClient", pendingClient);
206 
207  if (result != null) skipRemove = result.Value;
208 
209  if (!skipRemove && connectedClients.Count >= serverSettings.MaxPlayers)
210  {
211  RemovePendingClient(pendingClient, PeerDisconnectPacket.WithReason(DisconnectReason.ServerFull));
212  }
213 
214  if (IsPendingClientBanned(pendingClient, out string? banReason))
215  {
216  RemovePendingClient(pendingClient, PeerDisconnectPacket.Banned(banReason));
217  return;
218  }
219 
220  if (pendingClient.InitializationStep == ConnectionInitialization.Success)
221  {
222  NetworkConnection newConnection = pendingClient.Connection;
223  connectedClients.Add(newConnection);
224  pendingClients.Remove(pendingClient);
225 
226  callbacks.OnInitializationComplete.Invoke(newConnection, pendingClient.Name);
227 
228  CheckOwnership(pendingClient);
229  }
230 
231  pendingClient.TimeOut -= Timing.Step;
232  if (pendingClient.TimeOut < 0.0)
233  {
234  RemovePendingClient(pendingClient, PeerDisconnectPacket.WithReason(DisconnectReason.Timeout));
235  }
236 
237  if (Timing.TotalTime < pendingClient.UpdateTime) { return; }
238 
239  pendingClient.UpdateTime = Timing.TotalTime + 1.0;
240 
241  PeerPacketHeaders headers = new PeerPacketHeaders
242  {
243  DeliveryMethod = DeliveryMethod.Reliable,
244  PacketHeader = PacketHeader.IsConnectionInitializationStep | PacketHeader.IsServerMessage,
245  Initialization = pendingClient.InitializationStep
246  };
247 
248  INetSerializableStruct? structToSend = null;
249 
250  switch (pendingClient.InitializationStep)
251  {
252  case ConnectionInitialization.ContentPackageOrder:
253 
254  SerializableDateTime timeNow = SerializableDateTime.UtcNow;
255  structToSend = new ServerPeerContentPackageOrderPacket
256  {
257  ServerName = GameMain.Server.ServerName,
258  ContentPackages = ContentPackageManager.EnabledPackages.All
259  .Where(cp => cp.Files.Any())
260  .Where(cp => cp.HasMultiplayerSyncedContent || cp.Files.All(f => f is SubmarineFile))
261  .Select(contentPackage => new ServerContentPackage(contentPackage, timeNow))
262  .ToImmutableArray()
263  };
264 
265  break;
266  case ConnectionInitialization.Password:

271  };
