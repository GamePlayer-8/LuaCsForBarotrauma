105  private bool CheckInventory(Inventory inventory, Character character)
106  {
107  if (inventory == null) { return false; }
108  int count = 0;
109  HashSet<Item> eventTargets = new HashSet<Item>();
110  foreach (Identifier tag in itemTags)
111  {
112  foreach (var target in ParentEvent.GetTargets(tag))
113  {
114  if (target is Item item)
115  {
116  eventTargets.Add(item);
117  }
118  }
119  }
120  foreach (Item item in inventory.FindAllItems(it =>
121  itemTags.Any(it.HasTag) ||
122  itemIdentifierSplit.Contains(it.Prefab.Identifier) ||
123  eventTargets.Contains(it),
124  recursive: Recursive))
125  {
126  if (!ConditionalsMatch(item, character)) { continue; }
127  count++;
128  if (count >= Amount) { return true; }
129  }
130  return false;
131  }

133  private bool ConditionalsMatch(Item item, Character character = null)
134  {
135  if (item == null) { return false; }
136  foreach (PropertyConditional conditional in conditionals)
137  {
138  if (!conditional.Matches(item))
139  {
140  return false;
141  }
142  }
143  if (RequireEquipped)
144  {
145  if (character == null) { return false; }
146  return character.HasEquippedItem(item);
147  }
148  return true;
