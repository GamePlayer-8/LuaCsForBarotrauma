242  public Effect(ContentXElement element, string parentDebugName)
243  {

245 
246  resistanceFor = element.GetAttributeIdentifierArray("resistancefor", Array.Empty<Identifier>());
247  BlockTransformation = element.GetAttributeIdentifierArray("blocktransformation", Array.Empty<Identifier>());
248 
249  foreach (var subElement in element.Elements())
250  {
251  switch (subElement.Name.ToString().ToLowerInvariant())
252  {
253  case "statuseffect":
254  StatusEffects.Add(StatusEffect.Load(subElement, parentDebugName));
255  break;
256  case "statvalue":
257  var statType = CharacterAbilityGroup.ParseStatType(subElement.GetAttributeString("stattype", ""), parentDebugName);
258 
259  float defaultValue = subElement.GetAttributeFloat("value", 0f);
260  float minValue = subElement.GetAttributeFloat("minvalue", defaultValue);
261  float maxValue = subElement.GetAttributeFloat("maxvalue", defaultValue);
262 
263  AfflictionStatValues.TryAdd(statType, (minValue, maxValue));
264  break;
265  case "abilityflag":
266  var flagType = CharacterAbilityGroup.ParseFlagType(subElement.GetAttributeString("flagtype", ""), parentDebugName);
267  AfflictionAbilityFlags |= flagType;
268  break;
269  case "affliction":
270  DebugConsole.AddWarning($"Error in affliction \"{parentDebugName}\" - additional afflictions caused by the affliction should be configured inside status effects.");
271  break;
272  }
273  }

607  public LocalizedString GetDescription(float strength, Description.TargetType targetType)
608  {
609  foreach (var description in Descriptions)
610  {
611  if (strength < description.MinStrength || strength > description.MaxStrength) { continue; }

723  {
