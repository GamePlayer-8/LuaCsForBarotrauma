139  return

142  logError: false) ??

144  NPCSet.Get(NPCSetIdentifier,

148

150 
151  if (humanPrefab != null)
152  {
153  if (!AllowDuplicates &&
154  Character.CharacterList.Any(c => c.Info?.HumanPrefabIds.NpcIdentifier == NPCIdentifier && c.Info?.HumanPrefabIds.NpcSetIdentifier == NPCSetIdentifier))

163  {

166  newCharacter.TeamID = TeamID;
167  newCharacter.EnableDespawn = false;

169  if (LootingIsStealing)

171  foreach (Item item in newCharacter.Inventory.FindAllItems(recursive: true))
172  {
173  item.SpawnedInCurrentOutpost = true;
174  item.AllowStealing = false;
175  }
176  }

178  if (!TargetTag.IsEmpty && newCharacter != null)
179  {
180  ParentEvent.AddTarget(TargetTag, newCharacter);
181  }
182  spawnedEntity = newCharacter;
183  if (Level.Loaded?.StartOutpost?.Info is { } outPostInfo)
184  {

186  foreach (Identifier tag in humanPrefab.GetTags())
187  {
188  outPostInfo.AddOutpostNPCIdentifierOrTag(newCharacter, tag);
189  }
190  }
191  });
192  }
193  }
194  }
195  else if (!SpeciesName.IsEmpty)

213  }
214  }

326  public static WayPoint GetSpawnPos(SpawnLocationType spawnLocation, SpawnType? spawnPointType, IEnumerable<Identifier> moduleFlags = null, IEnumerable<Identifier> spawnpointTags = null, bool asFarAsPossibleFromAirlock = false, bool requireTaggedSpawnPoint = false)
327  {
328  bool requireHull = spawnLocation == SpawnLocationType.MainSub || spawnLocation == SpawnLocationType.Outpost;
329  List<WayPoint> potentialSpawnPoints = WayPoint.WayPointList.FindAll(wp => IsValidSubmarineType(spawnLocation, wp.Submarine) && (wp.CurrentHull != null || !requireHull));
330  potentialSpawnPoints = potentialSpawnPoints.FindAll(wp => wp.ConnectedDoor == null && wp.Ladders == null && wp.IsTraversable);
331  if (moduleFlags != null && moduleFlags.Any())
332  {
333  var spawnPoints = potentialSpawnPoints.Where(wp => wp.CurrentHull is Hull h && h.OutpostModuleTags.Any(moduleFlags.Contains));
334  if (spawnPoints.Any())
335  {
336  potentialSpawnPoints = spawnPoints.ToList();
337  }
338  }
339  if (spawnpointTags != null && spawnpointTags.Any())
340  {
341  var spawnPoints = potentialSpawnPoints.Where(wp => spawnpointTags.Any(tag => wp.Tags.Contains(tag) && wp.ConnectedDoor == null && wp.IsTraversable));
342  if (requireTaggedSpawnPoint || spawnPoints.Any())
343  {
344  potentialSpawnPoints = spawnPoints.ToList();
345  }
346  }
347  if (potentialSpawnPoints.None())
348  {
349  if (requireTaggedSpawnPoint && spawnpointTags != null && spawnpointTags.Any())
350  {
351  DebugConsole.NewMessage($"Could not find a spawn point for a SpawnAction (spawn location: {spawnLocation} (tag: {string.Join(",", spawnpointTags)}), skipping.", color: Color.White);
352  }
353  else
354  {
355  DebugConsole.ThrowError($"Could not find a spawn point for a SpawnAction (spawn location: {spawnLocation})");
356  }
357  return null;
358  }
359 
360  IEnumerable<WayPoint> validSpawnPoints;
361  if (spawnPointType.HasValue)
362  {
363  validSpawnPoints = potentialSpawnPoints.FindAll(wp => spawnPointType.Value.HasFlag(wp.SpawnType));
364  }
365  else
366  {
367  validSpawnPoints = potentialSpawnPoints.FindAll(wp => wp.SpawnType != SpawnType.Path);
368  if (!validSpawnPoints.Any()) { validSpawnPoints = potentialSpawnPoints; }
369  }
370 
371  //don't spawn in an airlock module if there are other options
372  var airlockSpawnPoints = potentialSpawnPoints.Where(wp => wp.CurrentHull?.OutpostModuleTags.Contains("airlock".ToIdentifier()) ?? false);
373  if (airlockSpawnPoints.Count() < validSpawnPoints.Count())
374  {
375  validSpawnPoints = validSpawnPoints.Except(airlockSpawnPoints);
376  }
377 
378  if (validSpawnPoints.None())
379  {
380  DebugConsole.ThrowError($"Could not find a spawn point of the correct type for a SpawnAction (spawn location: {spawnLocation}, type: {spawnPointType}, module flags: {((moduleFlags == null || !moduleFlags.Any()) ? "none" : string.Join(", ", moduleFlags))})");
381  return potentialSpawnPoints.GetRandomUnsynced();
382  }
383 
384  //avoid using waypoints if there's any actual spawnpoints available
385  if (validSpawnPoints.Any(wp => wp.SpawnType != SpawnType.Path))
386  {
387  validSpawnPoints = validSpawnPoints.Where(wp => wp.SpawnType != SpawnType.Path);
388  }
389 
390  //if not trying to spawn at a tagged spawnpoint, favor spawnpoints without tags
391  if (spawnpointTags == null || spawnpointTags.None())
392  {
393  var spawnPoints = validSpawnPoints.Where(wp => !wp.Tags.Any());
394  if (spawnPoints.Any())
395  {
396  validSpawnPoints = spawnPoints.ToList();
397  }
398  }
399 
400  if (asFarAsPossibleFromAirlock && airlockSpawnPoints.Any())
401  {
402  WayPoint furthestPoint = validSpawnPoints.First();
403  float furthestDist = 0.0f;
404  foreach (WayPoint waypoint in validSpawnPoints)
405  {
406  float dist = Vector2.DistanceSquared(waypoint.WorldPosition, airlockSpawnPoints.First().WorldPosition);
407  if (dist > furthestDist)
408  {
409  furthestDist = dist;
410  furthestPoint = waypoint;
411  }
412  }
413  return furthestPoint;
414  }
415  else
416  {
417  return validSpawnPoints.GetRandomUnsynced();
418  }
419  }
