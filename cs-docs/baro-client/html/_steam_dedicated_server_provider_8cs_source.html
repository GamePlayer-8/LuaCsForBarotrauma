49  private static void HandleResponsiveServer(Steamworks.Data.ServerInfo entry, Action<ServerInfo> onServerDataReceived)
50  {
51  TaskPool.Add($"QueryServerRules (GetServers, {entry.Name}, {entry.Address})", entry.QueryRulesAsync(),
52  t =>
53  {
54  if (t.Status == TaskStatus.Faulted)
55  {
56  TaskPool.PrintTaskExceptions(t, $"Failed to retrieve rules for {entry.Name}");
57  return;
58  }
59 
60  if (!t.TryGetResult(out Dictionary<string, string> rules)) { return; }
61  if (rules is null) { return; }
62  if (!InfoFromListEntry(entry).TryUnwrap(out var serverInfo)) { return; }
63  serverInfo.UpdateInfo(key =>
64  {
65  if (rules.TryGetValue(key, out var val)) { return val; }
66  return null;
67  });
68  serverInfo.Checked = true; //rules != null;
69 
70  onServerDataReceived(serverInfo);
71  });
72  }

84  protected override void RetrieveServersImpl(Action<ServerInfo> onServerDataReceived, Action onQueryCompleted)
85  {
86  if (!SteamManager.IsInitialized)
87  {
88  onQueryCompleted();
89  return;
90  }
91 
92  // All lambdas in here must only capture this call's
93  // query, not the provider's latest query
94  var selfServerQuery = new Steamworks.ServerList.Internet();
95  serverQuery = selfServerQuery;
96 
97  ConcurrentQueue<Steamworks.Data.ServerInfo> responsiveServers =
98  new ConcurrentQueue<Steamworks.Data.ServerInfo>();
99  ConcurrentQueue<Steamworks.Data.ServerInfo> unresponsiveServers =
100  new ConcurrentQueue<Steamworks.Data.ServerInfo>();
101 
102  selfServerQuery.OnResponsiveServer = responsiveServers.Enqueue;
103  selfServerQuery.OnUnresponsiveServer = unresponsiveServers.Enqueue;
104 
105  void dequeue(int? limit = null)
106  {
107  for (int i = 0; (!limit.HasValue || i < limit) && responsiveServers.TryDequeue(out var serverInfo); i++)
108  {
109  HandleResponsiveServer(serverInfo, onServerDataReceived);
110  }
111 
112  for (int i = 0; (!limit.HasValue || i < limit) && unresponsiveServers.TryDequeue(out var serverInfo); i++)
113  {
114  HandleUnresponsiveServer(serverInfo, onServerDataReceived);
115  }
116  }

118  IEnumerable<CoroutineStatus> dequeueCoroutine()
119  {
120  while (true)
121  {
122  dequeue(limit: 20);
123  yield return new WaitForSeconds(0.1f, ignorePause: true);
124  }
125  }
126  var selfQueryCoroutine = CoroutineManager.StartCoroutine(dequeueCoroutine(),
127  $"{nameof(SteamDedicatedServerProvider)}.{nameof(RetrieveServers)}.{nameof(dequeueCoroutine)}");
128  queryCoroutine = selfQueryCoroutine;
129 
130  TaskPool.Add("RunServerQuery", selfServerQuery.RunQueryAsync(timeoutSeconds: 30f),
131  t =>
132  {
133  try
134  {
135  // Clear the callbacks because it's too late now, we want to get this over with
136  selfServerQuery.OnResponsiveServer = null;
137  selfServerQuery.OnUnresponsiveServer = null;
138 
139  CoroutineManager.StopCoroutines(selfQueryCoroutine);
140  dequeue();
141 
142  if (t.Status == TaskStatus.Faulted) { TaskPool.PrintTaskExceptions(t, "Failed to retrieve servers"); }
143 
144  selfServerQuery.Dispose();
145  }
146  finally
147  {
148  onQueryCompleted();
149  }
150  });
