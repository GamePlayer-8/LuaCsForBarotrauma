113  public IEnumerable<Type> GetSubTypesInLoadedAssemblies<T>(bool rebuildList)
114  {
115  Type targetType = typeof(T);
116  string typeName = targetType.FullName ?? targetType.Name;
117 
118  // rebuild
119  if (rebuildList)
120  RebuildTypesList();
121 
122  // check cache
123  if (_subTypesLookupCache.TryGetValue(typeName, out var subTypeList))
124  {
125  return subTypeList;
126  }
127 
128  // build from scratch
129  OpsLockLoaded.EnterReadLock();
130  try
131  {
132  // build list
133  var list1 = _defaultContextTypes
134  .Where(kvp1 => targetType.IsAssignableFrom(kvp1.Value) && !kvp1.Value.IsInterface)
135  .Concat(LoadedACLs
136  .SelectMany(kvp => kvp.Value.AssembliesTypes)
137  .Where(kvp2 => targetType.IsAssignableFrom(kvp2.Value) && !kvp2.Value.IsInterface))
138  .Select(kvp3 => kvp3.Value)
139  .ToImmutableList();
140 
141  // only add if we find something
142  if (list1.Count > 0)
143  {
144  if (!_subTypesLookupCache.TryAdd(typeName, list1))
145  {
146  ModUtils.Logging.PrintError($"{nameof(AssemblyManager)}: Unable to add subtypes to cache of type {typeName}!");
147  }
148  }
149  else
150  {
151  ModUtils.Logging.PrintMessage($"{nameof(AssemblyManager)}: Warning: No types found during search for subtypes of {typeName}");
152  }
153 
154  return list1;
155  }
156  finally
157  {
158  OpsLockLoaded.ExitReadLock();
159  }

211  public IEnumerable<Type> GetTypesByName(string typeName)
212  {
213  List<Type> types = new();
214  if (typeName.IsNullOrWhiteSpace())
215  return types;
216 
217  bool byRef = false;
218  if (typeName.StartsWith("out ") || typeName.StartsWith("ref "))
219  {
220  typeName = typeName.Remove(0, 4);
221  byRef = true;
222  }
223 
224 
225  TypesListHelper();
226  if (types.Count > 0)
227  return types;
228 
229  // we couldn't find it, rebuild and try one more time
230  RebuildTypesList();
231  TypesListHelper();
232 
233  if (types.Count > 0)
234  return types;
235 
236  OpsLockLoaded.EnterReadLock();
237  try
238  {
239  // fallback to Type.GetType
240  Type t = Type.GetType(typeName, false, false);
241  if (t is not null)
242  {
243  types.Add(byRef ? t.MakeByRefType() : t);
244  return types;
245  }
246 
247  foreach (var assembly in AppDomain.CurrentDomain.GetAssemblies())
248  {
249  t = assembly.GetType(typeName, false, false);
250  if (t is not null)
251  types.Add(byRef ? t.MakeByRefType() : t);
252  }
253  }
254  finally
255  {
256  OpsLockLoaded.ExitReadLock();
257  }
258 
259  return types;

352  public AssemblyLoadingSuccessState LoadAssemblyFromMemory([NotNull] string compiledAssemblyName,
353  [NotNull] IEnumerable<SyntaxTree> syntaxTree,
354  IEnumerable<MetadataReference> externalMetadataReferences,
355  [NotNull] CSharpCompilationOptions compilationOptions,
356  string friendlyName,
357  ref Guid id,
358  IEnumerable<Assembly> externFileAssemblyRefs = null)
359  {
360  // validation
361  if (compiledAssemblyName.IsNullOrWhiteSpace())
362  return AssemblyLoadingSuccessState.BadName;
363 
364  if (!GetOrCreateACL(id, friendlyName, out var acl))
365  return AssemblyLoadingSuccessState.ACLLoadFailure;
366 
367  id = acl.Id; // pass on true id returned
368 
369  // this acl is already hosting an in-memory assembly
370  if (acl.Acl.CompiledAssembly is not null)
371  return AssemblyLoadingSuccessState.AlreadyLoaded;
372 
373  // compile
374  var state = acl.Acl.CompileAndLoadScriptAssembly(compiledAssemblyName, syntaxTree, externalMetadataReferences,
375  compilationOptions, out var messages, externFileAssemblyRefs);
376 
377  // get types
378  if (state is AssemblyLoadingSuccessState.Success)
379  {
380  _subTypesLookupCache.Clear();
381  acl.RebuildTypesList();
382  OnAssemblyLoaded?.Invoke(acl.Acl.CompiledAssembly);
383  }
384  else
385  {
386  ModUtils.Logging.PrintError($"Unable to compile assembly '{compiledAssemblyName}' due to errors: {messages}");
387  }
388 
389  return state;

416  public AssemblyLoadingSuccessState LoadAssembliesFromLocations([NotNull] IEnumerable<string> filePaths,
417  string friendlyName, ref Guid id)
418  {
419 
420  if (filePaths is null)
421  {
422  throw new ArgumentNullException(
423  $"{nameof(AssemblyManager)}::{nameof(LoadAssembliesFromLocations)}() | file paths supplied is null!");
424  }
425 
426  ImmutableList<string> assemblyFilePaths = filePaths.ToImmutableList(); // copy the list before loading
427

458  [MethodImpl(MethodImplOptions.NoInlining | MethodImplOptions.Synchronized)]
459  public bool TryBeginDispose()
460  {
461  OpsLockLoaded.EnterWriteLock();
462  OpsLockUnloaded.EnterWriteLock();
463  try
464  {
465  _subTypesLookupCache.Clear();
466 
467  foreach (KeyValuePair<Guid, LoadedACL> loadedAcl in LoadedACLs)
468  {
469  if (loadedAcl.Value.Acl is not null)
470  {
471  foreach (Delegate del in IsReadyToUnloadACL.GetInvocationList())
472  {
473  if (del is System.Func<LoadedACL, bool> { } func)
474  {
475  if (!func.Invoke(loadedAcl.Value))
476  return false; // Not ready, exit
477  }
478  }
479 
480  foreach (Assembly assembly in loadedAcl.Value.Acl.Assemblies)
481  {
482  OnAssemblyUnloading?.Invoke(assembly);
483  }
484 
485  UnloadingACLs.Add(new WeakReference<MemoryFileAssemblyContextLoader>(loadedAcl.Value.Acl, true));
486  loadedAcl.Value.ClearTypesList();
487  loadedAcl.Value.Acl.Unload();
488  OnACLUnload?.Invoke(loadedAcl.Value.Id);
489  }
490  }
491 
492  LoadedACLs.Clear();
493  return true;
494  }
495  catch
496  {
497  // should never happen
498  return false;
499  }
500  finally
501  {
502  OpsLockUnloaded.ExitWriteLock();
503  OpsLockLoaded.ExitWriteLock();
504  }
505  }

508  [MethodImpl(MethodImplOptions.NoInlining)]
509  public bool FinalizeDispose()
510  {
511  bool isUnloaded;
512  OpsLockUnloaded.EnterUpgradeableReadLock();
513  try
514  {
515  List<WeakReference<MemoryFileAssemblyContextLoader>> toRemove = new();
516  foreach (WeakReference<MemoryFileAssemblyContextLoader> weakReference in UnloadingACLs)
517  {
518  if (!weakReference.TryGetTarget(out _))
519  {
520  toRemove.Add(weakReference);
521  }
522  }
523 
524  if (toRemove.Any())
525  {
526  OpsLockUnloaded.EnterWriteLock();
527  try
528  {
529  foreach (WeakReference<MemoryFileAssemblyContextLoader> reference in toRemove)
530  {
531  UnloadingACLs.Remove(reference);
532  }
533  }
534  finally
535  {
536  OpsLockUnloaded.ExitWriteLock();
537  }
538  }
539  isUnloaded = !UnloadingACLs.Any();
540  }
541  finally
542  {
543  OpsLockUnloaded.ExitUpgradeableReadLock();
544  }
545 
546  return isUnloaded;

625  [MethodImpl(MethodImplOptions.NoInlining)]
626  private bool DisposeACL(Guid id)
627  {
628  OpsLockLoaded.EnterWriteLock();
629  OpsLockUnloaded.EnterWriteLock();
630  try
631  {
632  if (id.Equals(Guid.Empty) || !LoadedACLs.ContainsKey(id) || LoadedACLs[id] is null)
633  {
634  return false; // nothing to dispose of
635  }
636 
637  var acl = LoadedACLs[id];
638 
639  foreach (Assembly assembly in acl.Acl.Assemblies)
640  {
641  OnAssemblyUnloading?.Invoke(assembly);
642  }
643 
644  _subTypesLookupCache.Clear();
645  UnloadingACLs.Add(new WeakReference<MemoryFileAssemblyContextLoader>(acl.Acl, true));
646  acl.Acl.Unload();
647  OnACLUnload?.Invoke(acl.Id);
648 
649  return true;
650  }
651  catch
652  {
653  // should never happen
654  return false;
655  }
656  finally
657  {
658  OpsLockLoaded.ExitWriteLock();
659  OpsLockUnloaded.ExitWriteLock();

677  .ToImmutableDictionary(t => t.FullName ?? t.Name, t => t);

679  }
680  catch(ArgumentException _)

684  // some types must've had duplicate type names, build the list while filtering
685  Dictionary<string, Type> types = new();
686  foreach (var type in AssemblyLoadContext.Default.Assemblies.SelectMany(a => a.GetSafeTypes()))
687  {
688  try
689  {

691  }
692  catch
693  {
694  // ignore, null key exception
695  }
696  }
697 
698  _defaultContextTypes = types.ToImmutableDictionary();
699  }
700  catch (Exception e)
701  {
702  ModUtils.Logging.PrintError($"{nameof(AssemblyManager)}: Unable to create list of default assembly types! Default AssemblyLoadContext types searching not available.");
703 #if DEBUG
