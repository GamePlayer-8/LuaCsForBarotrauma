206  protected override void StartMissionSpecific(Level level)
207  {
208 #if SERVER
209  spawnInfo.Clear();
210 #endif
211  foreach (var target in targets)
212  {
213  bool usedExistingItem = false;
214  UInt16 originalInventoryID = 0;
215  byte originalItemContainerIndex = 0;
216  int originalSlotIndex = 0;
217  var executedEffectIndices = new List<(int listIndex, int effectIndex)>();
218 
219  target.Reset();
220  if (!IsClient)
221  {
222  //ruin/cave/wreck items are allowed to spawn close to the sub
223  float minDistance = target.SpawnPositionType switch
224  {
225  Level.PositionType.Ruin or
226  Level.PositionType.Cave or
227  Level.PositionType.Wreck or
228  Level.PositionType.Outpost => 0.0f,
229  _ => Level.Loaded.Size.X * 0.3f,
230  };
231  Vector2 position =
232  target.SpawnPositionType == Level.PositionType.None ?
233  Vector2.Zero :
234  Level.Loaded.GetRandomItemPos(target.SpawnPositionType, 100.0f, minDistance, 30.0f);
235 
236  if (!target.ExistingItemTag.IsEmpty)
237  {
238  var suitableItems = Item.ItemList.Where(it => it.HasTag(target.ExistingItemTag));
239  if (GameMain.GameSession?.Missions != null)
240  {
241  //don't choose an item that was already chosen as the target for another salvage mission
242  suitableItems = suitableItems.Where(it =>
243  GameMain.GameSession.Missions.None(m => m != this && m is SalvageMission salvageMission && salvageMission.targets.Any(t => t.Item == it)));
244  }
245  switch (target.SpawnPositionType)
246  {
247  case Level.PositionType.Cave:
248  case Level.PositionType.MainPath:
249  case Level.PositionType.SidePath:
250  target.Item = suitableItems.FirstOrDefault(it => Vector2.DistanceSquared(it.WorldPosition, position) < 1000.0f);
251 #if SERVER
252  usedExistingItem = target.Item != null;
253 #endif
254  break;
255  case Level.PositionType.Ruin:
256  case Level.PositionType.Wreck:
257  case Level.PositionType.Outpost:
258  foreach (Item it in suitableItems)
259  {
260  if (it.Submarine?.Info == null) { continue; }
261  if (target.SpawnPositionType == Level.PositionType.Ruin && it.Submarine.Info.Type != SubmarineType.Ruin) { continue; }
262  if (target.SpawnPositionType == Level.PositionType.Wreck && it.Submarine.Info.Type != SubmarineType.Wreck) { continue; }
263  if (target.SpawnPositionType == Level.PositionType.Outpost && it.Submarine.Info.Type != SubmarineType.Outpost) { continue; }
264  Rectangle worldBorders = it.Submarine.Borders;
265  worldBorders.Location += it.Submarine.WorldPosition.ToPoint();
266  if (Submarine.RectContains(worldBorders, it.WorldPosition))
267  {
268  target.Item = it;
269 #if SERVER
270  usedExistingItem = true;
271 #endif
272  break;
273  }
274  }
275  break;
276  default:
277  target.Item = suitableItems.FirstOrDefault();
278 #if SERVER
279  usedExistingItem = target.Item != null;
280 #endif
281  break;
282  }
283  }
284 
285  if (target.Item == null)
286  {
287  if (target.ItemPrefab == null && target.ContainerTag.IsEmpty)
288  {
289  DebugConsole.ThrowError($"Failed to find a target item for the mission \"{Prefab.Identifier}\". Item tag: {target.ExistingItemTag}");
290  continue;
291  }
292  target.Item = new Item(target.ItemPrefab, position, null);
293  target.Item.body.SetTransformIgnoreContacts(target.Item.body.SimPosition, target.Item.body.Rotation);
294  target.Item.body.FarseerBody.BodyType = BodyType.Kinematic;
295  }
296  else if (target.RequiredRetrievalState == Target.RetrievalState.Interact)
297  {
298  target.Item.OnInteract += () =>
299  {
300  target.Interacted = true;
301  };
302  }
303  for (int i = 0; i < target.StatusEffects.Count; i++)
304  {
305  List<StatusEffect> effectList = target.StatusEffects[i];
306  if (effectList.Count == 0) { continue; }
307  int effectIndex = Rand.Int(effectList.Count);
308  var selectedEffect = effectList[effectIndex];
309  target.Item.ApplyStatusEffect(selectedEffect, selectedEffect.type, deltaTime: 1.0f, worldPosition: target.Item.Position);
310 #if SERVER
311  executedEffectIndices.Add((i, effectIndex));
312 #endif
313  }
314 
315  target.Item.IsSalvageMissionItem = true;
316 
317  //try to find a container and place the item inside it
318  if (!target.ContainerTag.IsEmpty && target.Item.ParentInventory == null)
319  {
320  List<ItemContainer> validContainers = new List<ItemContainer>();
321  foreach (Item it in Item.ItemList)
322  {
323  if (!it.HasTag(target.ContainerTag)) { continue; }
324  if (!it.IsPlayerTeamInteractable) { continue; }
325  switch (target.SpawnPositionType)
326  {
327  case Level.PositionType.Cave:
328  case Level.PositionType.MainPath:
329  if (it.Submarine != null) { continue; }
330  break;
331  case Level.PositionType.Ruin:
332  if (it.Submarine?.Info == null || !it.Submarine.Info.IsRuin) { continue; }
333  break;
334  case Level.PositionType.Wreck:
335  if (it.Submarine?.Info == null || it.Submarine.Info.Type != SubmarineType.Wreck) { continue; }
336  break;
337  }
338  var itemContainer = it.GetComponent<ItemContainer>();
339  if (itemContainer != null && itemContainer.Inventory.CanBePut(target.Item)) { validContainers.Add(itemContainer); }
340  }
341  if (validContainers.Any())
342  {
343  var selectedContainer = validContainers.GetRandomUnsynced();
344  if (selectedContainer.Combine(target.Item, user: null))
345  {
346 #if SERVER
347  originalInventoryID = selectedContainer.Item.ID;
348  originalItemContainerIndex = (byte)selectedContainer.Item.GetComponentIndex(selectedContainer);
349  originalSlotIndex = target.Item.ParentInventory?.FindIndex(target.Item) ?? -1;
350 #endif
351  } // Placement successful
352  }
353  }
354  }
355 #if SERVER
356  spawnInfo.Add(
357  target,
358  new SpawnInfo(usedExistingItem, originalInventoryID, originalItemContainerIndex, originalSlotIndex, executedEffectIndices));
359 #endif
360  }
361  }

363  protected override void UpdateMissionSpecific(float deltaTime)
364  {
365  //make body dynamic when picked up
366  foreach (var target in targets)
367  {
368  var root = target.Item?.RootContainer ?? target.Item;
369  if (root == null) { continue; }
370  if (target.Item.ParentInventory != null && target.Item.body != null) { target.Item.body.FarseerBody.BodyType = BodyType.Dynamic; }
371  }
372 
373  if (IsClient) { return; }
374 
375  for (int i = 0; i < targets.Count; i++)
376  {
377  var target = targets[i];
378  if (i > 0 && !targets[i - 1].AllowContinueBeforeRetrieved && !targets[i - 1].Retrieved) { break; }
379  if (target.Item == null)
380  {
381 #if DEBUG
382  DebugConsole.ThrowError("Error in salvage mission " + Prefab.Identifier + " (item was null)");
383 #endif
384  return;
385  }
386  switch (target.State)
387  {
388  case Target.RetrievalState.None:
389  {
390  if (target.Interacted)
391  {
392  TrySetRetrievalState(Target.RetrievalState.Interact);
393  }
394  var root = target.Item?.RootContainer ?? target.Item;
395  if (root.ParentInventory?.Owner is Character character && character.TeamID == CharacterTeamType.Team1)
396  {
397  TrySetRetrievalState(Target.RetrievalState.PickedUp);
398  }
399  }
400 
401  break;
402  case Target.RetrievalState.PickedUp:
403  case Target.RetrievalState.RetrievedToSub:
404  {
405  Entity rootInventoryOwner = target.Item.GetRootInventoryOwner();
406  Submarine parentSub = target.Item.CurrentHull?.Submarine ?? rootInventoryOwner?.Submarine;
407 
408  bool inPlayerSub = parentSub != null && parentSub.Info.Type == SubmarineType.Player;
409  bool inPlayerInventory = false;
410  bool playerInFriendlySub = false;
411  if (rootInventoryOwner is Character character && character.TeamID == CharacterTeamType.Team1)
412  {
413  inPlayerInventory = true;
414  if (character.Submarine != null)
415  {
416  playerInFriendlySub =
417  character.IsInFriendlySub ||
418  (character.Submarine == Level.Loaded?.StartOutpost && Level.IsLoadedFriendlyOutpost && GameMain.GameSession?.Campaign.CurrentLocation is not { IsFactionHostile: true });
419  }
420  }
421 
422  if (inPlayerSub || (inPlayerInventory && playerInFriendlySub))
423  {
424  TrySetRetrievalState(Target.RetrievalState.RetrievedToSub);
425  }
426  else
427  {
428  target.State = Target.RetrievalState.PickedUp;
429  }
430  }
431  break;
432  }

434  void TrySetRetrievalState(Target.RetrievalState retrievalState)
435  {
436  if (retrievalState < target.State || target.State == retrievalState) { return; }
437  bool wasRetrieved = target.Retrieved;
438  target.State = retrievalState;
439  //increment the mission state if the target became retrieved
440  if (!wasRetrieved && target.Retrieved) { State = i + 1; }
441  }
442  }
443  if (targets.All(t => t.Retrieved))
444  {
445  State = targets.Count + 1;
446  }
447  }
