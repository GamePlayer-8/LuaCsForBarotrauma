61  private static async Task<Action> Initialize()
62  {

65  var failedToInitializeIntro = TextManager.Get("EosFailedToInitialize");
66 
67  if (EnableCrossplay is CrossplayChoice.Unknown)
68  {
69  // Don't even try to initialize EOS until we get the user's consent
70  return SteamAccountHasNoLinkedPuid();
71  }
72  if (EnableCrossplay is CrossplayChoice.Disabled)
73  {
74  // Crossplay is disabled, return immediately
75  return Success;

84  CrossThread.RequestExecutionOnMainThread(() => initResult = EosInterface.Core.Init(EosInterface.ApplicationCredentials.Client, enableOverlay: false));
85  if (initResult.TryUnwrapFailure(out var initError))
86  {

88  }

90  var steamPuidResult = await EosInterface.Login.LoginSteam();
91 
92  if (!steamPuidResult.TryUnwrapSuccess(out var steamPuidOrContToken))
93  {

95  }
96 
97  if (steamPuidOrContToken.TryGet(out EosInterface.ProductUserId puid))
