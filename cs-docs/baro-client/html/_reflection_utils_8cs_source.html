68  public static Option<TBase> ParseDerived<TBase, TInput>(TInput input) where TInput : notnull
69  {
70  static Option<TBase> none() => Option<TBase>.None();
71 
72  var derivedTypes = GetDerivedNonAbstract<TBase>();

74  Option<TBase> parseOfType(Type t)
75  {
76  //every TBase type is expected to have a method with the following signature:
77  // public static Option<T> Parse(TInput str)
78  var parseFunc = t.GetMethod("Parse", BindingFlags.Public | BindingFlags.Static);
79  if (parseFunc is null) { return none(); }
80 
81  var parameters = parseFunc.GetParameters();
82  if (parameters.Length != 1) { return none(); }
83 
84  var returnType = parseFunc.ReturnType;
85  if (!returnType.IsConstructedGenericType) { return none(); }
86  if (returnType.GetGenericTypeDefinition() != typeof(Option<>)) { return none(); }
87  if (returnType.GenericTypeArguments[0] != t) { return none(); }
88 
89  //some hacky business to convert from Option<T2> to Option<TBase> when we only know T2 at runtime
90  static Option<TBase> convert<T2>(Option<T2> option) where T2 : TBase
91  => option.Select(v => (TBase)v);
92  Func<Option<TBase>, Option<TBase>> f = convert;
93  var genericArgs = f.Method.GetGenericArguments();
94  genericArgs[^1] = t;
95  var constructedConverter =
96  f.Method.GetGenericMethodDefinition().MakeGenericMethod(genericArgs);
97 
98  return constructedConverter.Invoke(null, new[] { parseFunc.Invoke(null, new object[] { input }) })
99  as Option<TBase> ?? none();
100  }
101 
102  return derivedTypes.Select(parseOfType).FirstOrDefault(t => t.IsSome()) ?? none();
103  }
