65  public static TalentTreeStageState GetTalentOptionStageState(Character character, Identifier subTreeIdentifier, int index, List<Identifier> selectedTalents)
66  {
67  if (character?.Info?.Job.Prefab is null) { return TalentTreeStageState.Invalid; }
68 
69  if (!JobTalentTrees.TryGet(character.Info.Job.Prefab.Identifier, out TalentTree talentTree)) { return TalentTreeStageState.Invalid; }
70 
71  TalentSubTree subTree = talentTree.TalentSubTrees.FirstOrDefault(tst => tst.Identifier == subTreeIdentifier);
72 
73  if (subTree == null) { return TalentTreeStageState.Invalid; }
74 
75  TalentOption targetTalentOption = subTree.TalentOptionStages[index];
76 
77  if (targetTalentOption.TalentIdentifiers.Any(t => character.HasTalent(t)))
78  {
79  return TalentTreeStageState.Unlocked;
80  }
81 
82  if (targetTalentOption.TalentIdentifiers.Any(t => selectedTalents.Contains(t)))
83  {
84  return TalentTreeStageState.Highlighted;
85  }
86 
87  bool hasTalentInLastTier = true;
88  bool isLastTalentPurchased = true;
89 
90  int lastindex = index - 1;
91  if (lastindex >= 0)
92  {
93  TalentOption lastLatentOption = subTree.TalentOptionStages[lastindex];
94  hasTalentInLastTier = lastLatentOption.TalentIdentifiers.Any(HasTalent);
95  isLastTalentPurchased = lastLatentOption.TalentIdentifiers.Any(t => character.HasTalent(t));
96  }
97 
98  if (!hasTalentInLastTier)
99  {
100  return TalentTreeStageState.Locked;
101  }
102 
103  bool hasPointsForNewTalent = character.Info.GetTotalTalentPoints() - selectedTalents.Count > 0;
104 
105  if (hasPointsForNewTalent)
106  {
107  return isLastTalentPurchased ? TalentTreeStageState.Highlighted : TalentTreeStageState.Available;
108  }
109 
110  return TalentTreeStageState.Locked;

119  public static bool IsViableTalentForCharacter(Character character, Identifier talentIdentifier, ICollection<Identifier> selectedTalents)
120  {
121  if (character?.Info?.Job.Prefab == null) { return false; }
122  if (character.Info.GetTotalTalentPoints() - selectedTalents.Count() <= 0) { return false; }
123 
124  if (!JobTalentTrees.TryGet(character.Info.Job.Prefab.Identifier, out TalentTree talentTree)) { return false; }
125 
126  foreach (var subTree in talentTree.TalentSubTrees)
127  {
128  if (subTree.ForceUnlock && subTree.TalentOptionStages.Any(option => option.TalentIdentifiers.Contains(talentIdentifier))) { return true; }
129 
130  foreach (var talentOptionStage in subTree.TalentOptionStages)
131  {
132  bool hasTalentInThisTier = talentOptionStage.TalentIdentifiers.Any(t => selectedTalents.Contains(t));
133  if (!hasTalentInThisTier)
134  {
135  if (talentOptionStage.TalentIdentifiers.Contains(talentIdentifier))
136  {
137  return true;
138  }
139  else
140  {
141  break;
142  }
143  }
144  }
145  }
146 
147  return false;

150  public static List<Identifier> CheckTalentSelection(Character controlledCharacter, IEnumerable<Identifier> selectedTalents)
151  {
152  List<Identifier> viableTalents = new List<Identifier>();
153  bool canStillUnlock = true;
154  // keep trying to unlock talents until none of the talents are unlockable
155  while (canStillUnlock && selectedTalents.Any())
156  {
157  canStillUnlock = false;
158  foreach (Identifier talent in selectedTalents)
159  {
160  if (!viableTalents.Contains(talent) && IsViableTalentForCharacter(controlledCharacter, talent, viableTalents))
161  {
162  viableTalents.Add(talent);
163  canStillUnlock = true;
164  }
165  }
166  }
167  return viableTalents;
