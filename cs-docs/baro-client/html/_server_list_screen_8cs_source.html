146  {
147  if (Storage.IsNullOrEmpty()) { return; }
148  servers.Clear();
149  XDocument doc = XMLExtensions.TryLoadXml(Storage, out _);
150  if (doc?.Root is null) { return; }
151  servers.AddRange(doc.Root.Elements().Select(ServerInfo.FromXElement).NotNone().Distinct());
152  }

164  {
165  XDocument doc = new XDocument();

714  private void SortList(ColumnLabel sortBy, bool toggle)
715  {
716  if (!(labelHolder.GetChildByUserData(sortBy) is GUIButton button)) { return; }
717 
718  sortedBy = sortBy;

720  var arrowUp = button.GetChildByUserData("arrowup");
721  var arrowDown = button.GetChildByUserData("arrowdown");

729  child.GetChildByUserData("arrowdown").Visible = false;

736  ascending = !ascending;
