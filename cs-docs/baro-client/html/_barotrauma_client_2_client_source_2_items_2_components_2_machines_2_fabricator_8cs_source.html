406  missingIngredientCounts.Clear();
407 
408  FabricationRecipe targetItem = fabricatedItem ?? selectedItem;
409  if (targetItem != null)
410  {
411  foreach (FabricationRecipe.RequiredItem requiredItem in targetItem.RequiredItems)
412  {
413  if (missingIngredientCounts.ContainsKey(requiredItem))
414  {
415  missingIngredientCounts[requiredItem] += requiredItem.Amount;
416  }
417  else
418  {
419  missingIngredientCounts[requiredItem] = requiredItem.Amount;
420  }
421  }
422  foreach (Item item in inputContainer.Inventory.AllItems)
423  {
424  var missingIngredient = missingIngredientCounts.Keys.FirstOrDefault(mi => mi.MatchesItem(item));
425  if (missingIngredient == null) { continue; }
426 
427  if (missingIngredientCounts[missingIngredient] == 1)
428  {
429  missingIngredientCounts.Remove(missingIngredient);
430  }
431  else
432  {
433  missingIngredientCounts[missingIngredient]--;
434  }
435  }
436 
437  if (ingredientHighlightTimer <= 0.0f)
438  {
439  //highlight inventory slots that contain suitable ingredients in linked inventories
440  foreach (var inventory in linkedInventories)
441  {
442  if (inventory.visualSlots == null) { continue; }
443  for (int i = 0; i < inventory.Capacity; i++)
444  {
445  if (inventory.visualSlots[i].HighlightTimer > 0.0f) { continue; }
446  var availableItem = inventory.GetItemAt(i);
447  if (availableItem == null) { continue; }
448 
449  if (missingIngredientCounts.Keys.Any(it => it.MatchesItem(availableItem)))
450  {
451  inventory.visualSlots[i].ShowBorderHighlight(GUIStyle.Green, 0.5f, 0.5f, 0.2f);
452  continue;
453  }
454  if (availableItem.OwnInventory != null)
455  {
456  for (int j = 0; j < availableItem.OwnInventory.Capacity; j++)
457  {
458  var availableContainedItem = availableItem.OwnInventory.GetItemAt(i);
459  if (availableContainedItem == null) { continue; }
460  if (missingIngredientCounts.Keys.Any(it => it.MatchesItem(availableContainedItem)))
461  {
462  inventory.visualSlots[i].ShowBorderHighlight(GUIStyle.Green, 0.5f, 0.5f, 0.2f);
463  break;
464  }
465  }
466  }
467  }
468  }
469  ingredientHighlightTimer = 1.0f;
470  }
471 
472  int slotIndex = 0;
473  foreach (var kvp in missingIngredientCounts)
474  {
475  var requiredItem = kvp.Key;
476  int missingCount = kvp.Value;
477 
478  while (slotIndex < inputContainer.Capacity && inputContainer.Inventory.GetItemAt(slotIndex) != null)
479  {
480  slotIndex++;
481  }
482 
483  if (slotIndex >= inputContainer.Capacity) { break; }
484 
485  if (slotIndex < inputContainer.Capacity &&
486  inputContainer.Inventory.visualSlots[slotIndex].HighlightTimer <= 0.0f &&
487  availableIngredients.Any(i => i.Value.Any() && requiredItem.MatchesItem(i.Value.First())))

500 
501  if (missingCount > 1)
502  {
503  Vector2 stackCountPos = new Vector2(slotRect.Right, slotRect.Bottom);
504  string stackCountText = "x" + missingCount;
505  stackCountPos -= GUIStyle.SmallFont.MeasureString(stackCountText) + new Vector2(4, 2);
506  GUIStyle.SmallFont.DrawString(spriteBatch, stackCountText, stackCountPos + Vector2.One, Color.Black);

512  DrawConditionBar(spriteBatch, requiredItem.MinCondition);
